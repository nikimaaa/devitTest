cd var/www/html # Перехожу в папку html
ll # Вывожу содержимое папки html

grep 404 404.html # Ищу все вхождения строки "404" в файле 404.html

sed 's/404/403/gw 403.html' 404.html 
# Меняю все вхождения строки "404" в файле 404.html на "403" и вывожу результат в файл 403.html.
# s - флаг обозначающий замену строк
# 404 - строка для поиска
# 403 - строка для замены
# g - провести замену для всех найденых строк
# w - результат записать в новый файл
# 403.html - файл для записи результата
# 404.html - файл откуда берутся данные

chown nikima index.html # присваиваю права владельца пользователю
chown .www-data index.html # присваиваю права владельца группе
chmod u=rw- index.html # Разрешаю пользователю чтение и запись файла (u - пользователь, r - права на чтение, w - права на запись)
chmod g=r-- index.html # Разрешаю группе чтение файла (g - группа, r - права на чтение)
chmod o=--- index.html 
# Запрещаю всем пользователям которые не состоят в группе или не являются владельцами доступ к файлу (o - остальные пользователи)

ln -s ../../access.log access.log # Создаю символическую ссылку. Флаг -s показівает что ссылка должна быть символической

ls -R ../../  # Вывожу все файлы в папке var. Флаг -R позволяет рекурсивно вывести содержимое всех подкаталогов var

nano 503.html 
# Открываю через редактор nano файл 503.html. После добавления необходимого текста сохраняю данные комбинацией Ctrl+S и возвращаюсь в терминал Ctrl+X
